 * Handle offline test runners
 * Reap offline test runners that have not checked in for more than
 * 'sn_atf.runner.offline_retention.timeout' days
 */
function handleOfflineTestRunners() {
    var offlineRetentionDays = GlideProperties.getInt("sn_atf.runner.offline_retention.timeout", 1);

        offlineRetentionDays = 1;
    var gr = new GlideRecord("sys_atf_agent");

    // if last_checkin date is older than offlineRetentionDays ago date
    gr.addQuery("last_checkin", "<", gs.daysAgo(offlineRetentionDays));

    gr.query();
    gr.deleteMultiple();
}

/*
 * Mark all test runners that have not checked in with a heartbeat in the last
 * 'sn_atf.runner.heartbeat.timeout' seconds as offline
 */
function handleOnlineTestRunners() {
    // Handle online test runners
    var testRunnerTimeout = GlideProperties.getInt("sn_atf.runner.heartbeat.timeout", 120);
    if (testRunnerTimeout <= 0 || testRunnerTimeout > 1800)
        testRunnerTimeout = 120;

    gr.addQuery("last_checkin", "<", gs.minutesAgo(testRunnerTimeout/60));
    gr.addQuery("status", "online");
    gr.query();
    while (gr.next()) {
        gr.status = "offline";
        gr.status_reason = gs.getMessage("Client test runner did not check in");
        gr.update();
    }
}
